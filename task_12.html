    posts = response.json()

# Filter posts by user ID
user_id = 1
filtered_posts = [post for post in posts if post['userId'] == user_id]

# Sort comments by email

# Aggregate albums by user ID
album_dict = {}
for album in albums:
    user_id = album['userId']
    if user_id not in album_dict:
        album_dict[user_id] = [album]

for post in filtered_posts:

print("Comments sorted by email:")
for comment in sorted_comments:

    print(f"Email: {comment['email']}")
    print(f"Body: {comment['body']}\n")

print("Albums aggregated by user ID:")
